string topological_sort(list <int>* ls, int k){
char arr[k];
stack <int> s;
set <int> st;
int ind = k - 1;
for (int i = k - 1; i >= 0; i--) {
if (st.find(i) == st.end()) {
s.push(i);
st.insert(i);
while (!s.empty()) {
int p = s.top();
list <int>::iterator it;
int temp = 0;
for (it = ls[p].begin(); it != ls[p].end(); it++) {
if (st.find(*it) == st.end()) {
st.insert(*it);
s.push(*it);
temp = 1;
}
}
if (temp == 0) {
char ch = (char)(p + 'A');
arr[ind] = ch;
ind--;
s.pop();
}
}
}
}
return arr;
}
